    <script>
        // --- تهيئة الأيقونات ---
        lucide.createIcons();

        // --- إعدادات Firebase الخاصة بـ Rama ---
        const firebaseConfig = {
            apiKey: "AIzaSyAHBjSDDwbcXx0MEWO_59nN26Vj__58caA",
            databaseURL: "https://smarthall-d1a00-default-rtdb.firebaseio.com/",
            projectId: "smarthall-d1a00"
        };

        // تهيئة Firebase
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // ربط الحالة بـ Firebase وتغيير الواجهة فوراً
        const statusBadge = document.getElementById('status-badge');
        const statusDot = document.getElementById('status-dot');
        const statusText = document.getElementById('status-text');

        // استماع لحالة القاعة AF-10 من ESP32
        db.ref('AF10_Status').on('value', (snapshot) => {
            const isOccupied = snapshot.val();
            updateHall('AF10', isOccupied);
            
            // تحديث حالة الاتصال بالأعلى
            statusDot.className = "w-2.5 h-2.5 bg-emerald-500 rounded-full animate-ping";
            statusText.innerText = "متصل ومزامن الآن";
            statusBadge.className = "bg-emerald-500/10 text-emerald-400 px-6 py-3 rounded-2xl border border-emerald-500/20 text-sm font-bold flex items-center gap-3 transition-all";
        }, (error) => {
            statusDot.className = "w-2.5 h-2.5 bg-rose-500 rounded-full";
            statusText.innerText = "خطأ في الاتصال";
        });

        // نظام القفل والتحقق
        let pin = "";
        const correctPin = "8427";

        const keypad = document.getElementById('keypad');
        const keys = ['1','2','3','4','5','6','7','8','9','C','0','✓'];

        keys.forEach(key => {
            const btn = document.createElement('button');
            btn.innerText = key;
            btn.className = "py-6 glass text-white text-2xl font-black rounded-[1.5rem] key-btn";
            if(key === 'C') btn.classList.add('text-rose-400');
            if(key === '✓') btn.classList.add('text-emerald-400');
            btn.onclick = () => handleInput(key);
            keypad.appendChild(btn);
        });

        function handleInput(key) {
            const msg = document.getElementById('error-msg');
            msg.innerText = "";

            if(key === 'C') pin = "";
            else if(key === '✓') {
                if(pin === correctPin) {
                    document.getElementById('lock-screen').classList.add('opacity-0', 'scale-95', 'pointer-events-none');
                    setTimeout(() => {
                        document.getElementById('lock-screen').classList.add('hidden');
                        document.getElementById('app-screen').classList.remove('hidden');
                    }, 500);
                } else {
                    msg.innerText = "الرمز السري غير صحيح";
                    pin = "";
                }
            } else if(pin.length < 4) {
                pin += key;
            }
            updateDots();
        }

        function updateDots() {
            const dots = document.querySelectorAll('#dots div');
            dots.forEach((dot, i) => {
                if(i < pin.length) {
                    dot.className = "w-4 h-4 rounded-full bg-indigo-500 shadow-[0_0_15px_#6366f1]";
                } else {
                    dot.className = "w-4 h-4 rounded-full bg-slate-700";
                }
            });
        }

        function updateHall(id, occupied) {
            const header = document.getElementById('header-' + id);
            const badge = document.getElementById('badge-' + id);
            const sensorText = document.getElementById('sensor-' + id);
            const iconContainer = document.getElementById('icon-' + id);
            const pulse = document.getElementById('pulse-' + id);

            if(occupied) {
                header.className = "bg-rose-600 p-10 text-white transition-all duration-700";
                badge.innerText = "مشغولة حالياً";
                sensorText.innerText = "حركة مسجلة داخل القاعة";
                sensorText.className = "font-black text-lg text-rose-400";
                iconContainer.innerHTML = '<i data-lucide="user-x" class="w-8 h-8"></i>';
                pulse.classList.remove('hidden');
            } else {
                header.className = (id === 'AF10' ? "bg-emerald-600" : "bg-amber-600") + " p-10 text-white transition-all duration-700";
                badge.innerText = (id === 'AF10' ? "شاغرة حالياً" : "بانتظار المحاضرة");
                sensorText.innerText = "لا توجد حركة مسجلة";
                sensorText.className = "font-black text-lg text-slate-300";
                iconContainer.innerHTML = `<i data-lucide="${id === 'AF10' ? 'check-circle' : 'clock'}" class="w-8 h-8"></i>`;
                pulse.classList.add('hidden');
            }
            lucide.createIcons();
        }
    </script>
