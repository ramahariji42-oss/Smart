    <script>
        // Initialize Icons
        lucide.createIcons();

        // Live Clock logic
        setInterval(() => {
            const now = new Date();
            document.getElementById('live-clock').innerText = now.toLocaleTimeString('ar-SA');
        }, 1000);

        let activeDoctor = "";

        // Login Simulation
        function handleLogin() {
            const input = document.getElementById('doc-input');
            if (input.value.trim().length < 3) {
                alert("يرجى إدخال اسمك الحقيقي دكتور.");
                return;
            }
            activeDoctor = input.value;
            document.getElementById('login-form').classList.add('hidden');
            document.getElementById('logged-in-user').classList.remove('hidden');
            document.getElementById('welcome-msg').innerText = "مرحباً د. " + activeDoctor;
        }

        function handleLogout() {
            activeDoctor = "";
            document.getElementById('login-form').classList.remove('hidden');
            document.getElementById('logged-in-user').classList.add('hidden');
        }

        // Booking Simulation
        function book(hallId) {
            if (!activeDoctor) {
                alert("يرجى تسجيل الدخول باسمك دكتور أولاً لتتمكن من الحجز.");
                return;
            }

            const header = document.getElementById('header-' + hallId);
            const label = document.getElementById('label-' + hallId);
            const desc = document.getElementById('desc-' + hallId);
            const badge = document.getElementById('booking-badge-' + hallId);
            const nameSpan = document.getElementById('reserved-' + hallId);
            const btn = document.getElementById('btn-' + hallId);
            const icon = document.getElementById('icon-' + hallId);

            // Simulate database update visual
            header.classList.remove('bg-emerald-500');
            header.classList.add('bg-indigo-600');
            
            label.innerText = "تم الحجز بنجاح";
            desc.innerText = "لقد قمت بحجز هذه القاعة، يرجى التوجه إليها الآن وتفعيل السويتش.";
            
            nameSpan.innerText = activeDoctor;
            badge.classList.remove('hidden');
            
            btn.disabled = true;
            btn.innerHTML = '<i data-lucide="check"></i> تم حجزها بواسطتك';
            
            icon.setAttribute('data-lucide', 'shield-check');
            lucide.createIcons();
            
            // Celebration Alert
            alert("تم حجز قاعة " + hallId + " بنجاح باسم د. " + activeDoctor);
        }
    </script>
